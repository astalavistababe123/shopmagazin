                    nextBtn.click();

                } else if (touchEndX - touchStartX > 50) { // Swiped right

                    prevBtn.click();

                }

            });

            function updateConfigurator(phone) {

                phoneName.textContent = phone.name;

                phoneImage.src = phone.image;

                colorSelector.innerHTML = '';

                Object.entries(phone.colors).forEach(([name, hex], index) => {

                    const colorChip = document.createElement('div');

                    colorChip.className = 'chip color-chip';

                    colorChip.style.backgroundColor = hex;

                    colorChip.dataset.colorName = name;

                    if (index === 0) colorChip.classList.add('active');

                    colorSelector.appendChild(colorChip);

                });

            }

                const selectedPhone = phoneDatabase.find(p => p.id === phoneId);

                if (selectedPhone) openModal(selectedPhone);

            });

                    e.target.classList.add('active');

                });

            }
